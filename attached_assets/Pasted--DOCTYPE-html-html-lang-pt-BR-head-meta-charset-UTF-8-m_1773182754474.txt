        function shootConfetti() {
            const colors = ['#FF7EB3', '#42D392', '#38BDF8', '#FDE68A', '#FF6B6B'];
            for(let i = 0; i < 50; i++) {
                let conf = document.createElement('div');
                conf.className = 'confetti';
                conf.style.left = Math.random() * 100 + 'vw';
                conf.style.backgroundColor = colors[Math.floor(Math.random() * colors.length)];
                conf.style.width = (Math.random() * 10 + 5) + 'px';
                conf.style.height = (Math.random() * 10 + 5) + 'px';
                conf.style.borderRadius = Math.random() > 0.5 ? '50%' : '0';
                conf.style.animationDuration = (Math.random() * 2 + 2) + 's';
                conf.style.animationDelay = (Math.random() * 0.5) + 's';
                document.body.appendChild(conf);
                
                // Limpa o confete depois que ele cai
                setTimeout(() => conf.remove(), 4000);
            }
        }

        // --- MÓDULO 7: O DESAFIO REVERSO ---
        let chalTarget = 5;
        let chalDots = { left: 0, right: 0 };
        const chalEqDisplay = document.getElementById('chalEquation');

        function changeTarget(amount) {
            let newVal = chalTarget + amount;
            if (newVal >= 1 && newVal <= 18) { // 18 é o máximo do dominó (9+9)
                chalTarget = newVal;
                document.getElementById('chalTargetNum').innerText = chalTarget;
                checkChallengeWin();
            }
        }

        function updateChalDomino(side, amount) {
            let newVal = chalDots[side] + amount;
            if (newVal >= 0 && newVal <= 9) {
                chalDots[side] = newVal;
                renderChalDomino(side);
                checkChallengeWin();
            }
        }

        function renderChalDomino(side) {
            const container = document.getElementById(side === 'left' ? 'chalDomLeft' : 'chalDomRight');
            container.innerHTML = '';
            for (let i = 0; i < chalDots[side]; i++) {
                const dot = document.createElement('div');
                dot.classList.add('dot');
                container.appendChild(dot);
            }
        }

        function checkChallengeWin() {
            let total = chalDots.left + chalDots.right;
            if (total === chalTarget) {
                // ACERTOU!
                chalEqDisplay.innerText = `${chalDots.left} + ${chalDots.right} = ${total} 🎉`;
                chalEqDisplay.classList.add('success-glow');
                shootConfetti();
                
                // Remove o brilho depois de um tempinho
                setTimeout(() => chalEqDisplay.classList.remove('success-glow'), 2000);
            } else {
                // Ainda não
                if(chalDots.left === 0 && chalDots.right === 0) {
                     chalEqDisplay.innerText = `? + ? = ${chalTarget}`;
                } else {
                     chalEqDisplay.innerText = `${chalDots.left} + ${chalDots.right} = ${total}`;
                }
                chalEqDisplay.classList.remove('success-glow');
            }
        }
        
        // Inicializa o dominó do desafio vazio
        renderChalDomino('left');
        renderChalDomino('right');
        checkChallengeWin();

            const start = calcVal1; const op = calcOp; const steps = calcVal2; const endNum = op === '+' ? start + steps : start - steps;
            if (endNum < 0) { eq3D.innerText = "Opa! O bonequinho não sabe andar para antes do zero ainda! 🚫"; return; }
            
            buildTrack(Math.max(15, start + steps + 5));
            let currentPos = start; let currentStepCount = 0;
            char3D.style.transition = 'none'; char3D.style.left = (currentPos * TILE_WIDTH) + 'px'; charBubble.style.opacity = 0; eq3D.innerText = `Posição inicial: ${start}`;

            setTimeout(() => {
                char3D.style.transition = 'left 0.5s cubic-bezier(0.34, 1.56, 0.64, 1), transform 0.2s';
                function takeStep() {
                    if (currentStepCount >= steps) {
                        eq3D.innerText = `${start} ${op} ${steps} = ${endNum}! 🎉`; charBubble.innerText = "Cheguei!"; char3D.style.transform = "rotateX(-50deg) translateZ(40px) translateY(-30px) scale(1.2)"; showLesson(start, op, steps, endNum); return;
                    }
                    currentStepCount++; if (op === '+') currentPos++; else currentPos--;
                    char3D.style.left = (currentPos * TILE_WIDTH) + 'px'; sceneContainer.scrollTo({ left: (currentPos * TILE_WIDTH) - 100, behavior: 'smooth' });
                    const currentTile = document.getElementById('tile-'+currentPos); currentTile.classList.add('passed');
                    const marker = document.createElement('div'); marker.className = 'step-marker'; marker.innerText = `${op}1`; currentTile.appendChild(marker);
                    charBubble.style.opacity = 1; charBubble.innerText = `${op}${currentStepCount}`; eq3D.innerText = `Andando: ${op}${currentStepCount} (Estamos no ${currentPos})`;
                    walkTimeout = setTimeout(takeStep, 900); 
                }
                if (steps === 0) takeStep(); else takeStep();
            }, 500);
        }
